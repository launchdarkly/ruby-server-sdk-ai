def to_h
  {
    _ldMeta: {
      enabled: @enabled || false,
    },
    messages: @messages.is_a?(Array) ? @messages.map { |msg| msg&.to_h } : nil,
    model: @model&.to_h,
    provider: @provider&.to_h,
  }
end
